# Initialize an empty list to store tasks todo_list = [] # Function to add a task to the to-do list def add_task(task): todo_list.append(task) print(f"Task '{task}' added to the to-do list.") # Function to remove a task from the to-do list def remove_task(task): if task in todo_list: todo_list.remove(task) print(f"Task '{task}' removed from the to-do list.") else: print(f"Task '{task}' not found in the to-do list.") # Function to display the to-do list def display_todo_list(): if not todo_list: print("Your to-do list is empty.") else: print("To-Do List:") for index, task in enumerate(todo_list, start=1): print(f"{index}. {task}") # Main loop while True: print("\nOptions:") print("1. Add a task") print("2. Remove a task") print("3. Display to-do list") print("4. Quit") choice = input("Enter your choice (1/2/3/4): ") if choice == "1": task = input("Enter the task you want to add: ") add_task(task) elif choice == "2": task = input("Enter the task you want to remove: ") remove_task(task) elif choice == "3": display_todo_list() elif choice == "4": print("Goodbye!") break else: print("Invalid choice. Please choose a valid option (1/2/3/4).")
